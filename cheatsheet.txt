ListNode* reverseList(ListNode* head)
{
    if(( head == nullptr ) || ( head->next == nullptr ))
        return head;
    
    ListNode *prev = nullptr;
    ListNode *curr = head;

    while( curr != nullptr )
    {
        ListNode *ntemp = curr->next;
        curr->next = prev;
        prev = curr;
        curr = ntemp;
    }
    return prev;
}
